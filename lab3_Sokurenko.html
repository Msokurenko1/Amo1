def bubble(nums):
    swapped = True
    a = 0
    while swapped:
        swapped = False
        for i in range(len(nums) - a - 1):
            a += 1
            if nums[i] > nums[i + 1]:
                nums[i], nums[i + 1] = nums[i + 1], nums[i]
                swapped = True

def insertion_sort(nums):

        key = nums[j]
        i = j - 1
        while i >= 0 and nums[i] > key:
            nums[i + 1] = nums[i]
            i -= 1
        nums[i + 1] = key
